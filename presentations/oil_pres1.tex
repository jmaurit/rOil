%--- Benchmark modelling formula ----------------%
\begin{frame}[plain]
	\begin{multline}
	\nonumber Log(Production_{i,t})=f(time\_to\_peak_{i,t}, total\_recoverable\_oil_i) \\
	+ f(peak\_to\_end_{i,t}, total\_recoverable\_oil_i)
	+ f(year_{i,t}) + \epsilon
	\end{multline}

	\begin{equation}
		\nonumber \epsilon \sim Normal(0, \sigma^2)
		\end{equation}
\end{frame}
